// Program otevre jedno hlavni okno a vykresli strom. Je predveden zasobnik
// matic. Program se ukonci stiskem klavesy ESC, klavesami
// [,] (carka) a [.] (tecka) lze stromem rotovat, klavesami [<] a [>]
// lze menit uhel rozvetveni stromu.
//---------------------------------------------------------------------

#include <GL/glut.h>                            // hlavickovy soubor funkci GLUTu a OpenGL


int angle=0;                                    // uhel, o ktery se budou trojuhelniky rotovat
int delta=30;                                   // uhel vetveni



//---------------------------------------------------------------------
// Nastaveni souradneho systemu v zavislosti na velikosti okna
//---------------------------------------------------------------------
void onResize(int w, int h)                     // argumenty w a h reprezentuji novou velikost okna
{
    glViewport(0, 0, w, h);                     // viditelna oblast pres cele okno
    glMatrixMode(GL_PROJECTION);                // zacatek modifikace projekcni matice
    glLoadIdentity();                           // vymazani projekcni matice (=identita)
    glOrtho(0, w, 0, h, -1, 1);                 // mapovani abstraktnich souradnic do souradnic okna
}



//---------------------------------------------------------------------
// Tato funkce je volana pri kazdem prekresleni okna
// Po zavolani teto funkce je vykreslen transformovany stromecek
//---------------------------------------------------------------------
void onDisplay(void)
{
    glClearColor(0.0, 0.0, 0.0, 0.0);           // nastaveni mazaci barvy na cernou
    glClear(GL_COLOR_BUFFER_BIT);               // vymazani vsech bitovych rovin barvoveho bufferu

    glColor3f(1.0f, 1.0f, 1.0f);                // bila barva zobrazeni objektu
    glLineWidth(3.0f);                          // sirka usecek je 3 pixely
    glEnable(GL_LINE_SMOOTH);                   // povoleni antialiasingu usecek
    glMatrixMode(GL_MODELVIEW);                 // bude se meni modelova matice

    glLoadIdentity();                           // nastavit jednotkovou matici
    glTranslatef(200.0f, 200.0f, 0.0f);         // posun pocatku vykreslovani objektu
    glRotatef(angle, 0.0f, 0.0f, 1.0f);         // rotace o zadany uhel

    glBegin(GL_LINES);                          // vykresleni kmene
        glVertex2i(0,0);
        glVertex2i(0,100);
    glEnd();

    glTranslatef(0.0f, 100.0f, 0.0f);           // posun na konec kmene

    glPushMatrix();                             // ulozit transformacni matici
    glRotatef((GLfloat)delta, 0.0f, 0.0f, 1.0f);// rozvetveni
    glBegin(GL_LINES);                          // kresba prvni vetve
        glVertex2i(0,0);
        glVertex2i(0,100);
    glEnd();
    glPopMatrix();                              // obnovit transformacni matici

    glRotatef((GLfloat)-delta, 0.0f, 0.0f, 1.0f);// rozvetveni
    glBegin(GL_LINES);                          // kresba druhe vetve
        glVertex2i(0,0);
        glVertex2i(0,100);
    glEnd();

    glFlush();                                  // provedeni a vykresleni zmen
}



//---------------------------------------------------------------------
// Tato funkce je volana pri stlaceni ASCII klavesy
//---------------------------------------------------------------------
void onKeyboard(unsigned char key, int x, int y)
{
    switch (key) {
        case 27:    exit(0);    break;          // pokud byla stlacena klavesa ESC, konec programu
        case ',':   angle++;    glutPostRedisplay(); break; // rotace na klavesu [,]
        case '.':   angle--;    glutPostRedisplay(); break; // rotace na klavesu [.]
        case '<':   delta--;    glutPostRedisplay(); break; // zmena vetveni klavesou [<]
        case '>':   delta++;    glutPostRedisplay(); break; // zmena vetveni klavesou [>]
        default:    break;
    }
    if (key==27) exit(0);                       // pokud byla stlacena klavesa ESC, konec programu

}



//---------------------------------------------------------------------
// Hlavni funkce konzolove aplikace
//---------------------------------------------------------------------
int main(int argc, char **argv)
{
    glutInit(&argc, argv);                      // inicializace knihovny GLUT
    glutCreateWindow("Priklad na OpenGL cislo 9");// vytvoreni okna pro kresleni
    glutReshapeWindow(400, 400);                // zmena velikosti okna
    glutPositionWindow(100, 100);               // pozice leveho horniho rohu okna
    glutDisplayFunc(onDisplay);                 // registrace funkce volane pri prekreslovani okna
    glutReshapeFunc(onResize);                  // registrace funkce volane pri zmene velikosti okna
    glutKeyboardFunc(onKeyboard);               // registrace funkce volani pri stlaceni klavesy
    glutMainLoop();                             // nekonecna smycka, kde se volaji zaregistrovane funkce
    return 0;                                   // navratova hodnota vracena operacnimu systemu
}
