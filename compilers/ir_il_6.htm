1. Technologie mezijazyků (mezikódů) a bajtkódů v moderních interpretrech a překladačích: LLVM IR a SSA

V předchozí pětici článků o mezijazyku (mezikódu) LLVM IR [1] [2] [3] [4] [5] jsme se seznámili s prakticky všemi instrukcemi, které jsou v LLVM IR definovány. Prozatím jsme se na tento mezijazyk dívali jako na určitou formu pokročilejšího assembleru, který (na rozdíl) od většiny běžných assemblerů používá typové informace a navíc obsahuje relativně ortogonální instrukční sadu. Poměrně dobrým příkladem je instrukce součtu, která je použitelná jak pro skalární operandy, tak i pro součet vektorů atd. Ve skutečnosti je však LLVM IR navržen sofistikovanějším způsobem; konkrétně je navržen tak, aby se nad mezikódem daly provádět různé optimalizační operace. Proto se v tomto mezijazyku používá forma SSA neboli Static single-assignment, s jejímž konceptem se seznámíme v dnešním článku.

A se SSA souvisí i instrukce pojmenovaná phi, kterou jsme sice ve výpisech LLVM IR už několikrát viděli, ale prozatím jsme ji do jisté míry ignorovali. Název této instrukce je odvozen od takzvaných Φ-funkcí, což je označení, které možná souvisí s klíčovým slovem if (ovšem to je pouze domněnka autora článku). Dnes se s touto instrukcí seznámíme podrobněji, čímž vlastně dokončíme i celý popis LLVM IR (resp. přesněji řečeno jeho relevantních částí).

unsigned int gcd(unsigned int u, unsigned int v) {
    while (v) {
        int t = u;
        u = v;
        v = t % v;
    }
    return u;
}

; Function Attrs: nofree norecurse nosync nounwind optsize memory(none) uwtable
define dso_local i32 @gcd(i32 noundef %0, i32 noundef %1) local_unnamed_addr #0 {

unsigned int numeric_value(unsigned int x, unsigned int y) {
    switch (x) {
    case 0:
        y++;
        break;
    case 1:
        y += 2;
        break;
    case 2:
        y -= 2;
        break;
    case 3:
        y--;
        break;
    }
    return y;
}

long fibonacciIter(int n) {
    if (n <= 1) {
        return n;
    }
    long result = 0;
    long n1 = 0;
    long n2 = 1;
    for (n--; n > 0; n--) {
        result = n1 + n2;
        n1 = n2;
        n2 = result;
    }
    return result;
}

; Function Attrs: nofree norecurse nosync nounwind optsize memory(none) uwtable
define dso_local i64 @fibonacciIter(i32 noundef %0) local_unnamed_addr #0 {
  %2 = icmp slt i32 %0, 2
  br i1 %2, label %3, label %5

3:                                                ; preds = %1
  %4 = sext i32 %0 to i64
  br label %12

5:                                                ; preds = %1, %5
  %6 = phi i64 [ %10, %5 ], [ 1, %1 ]
  %7 = phi i64 [ %6, %5 ], [ 0, %1 ]
  %8 = phi i32 [ %9, %5 ], [ %0, %1 ]
  %9 = add nsw i32 %8, -1
  %10 = add nsw i64 %6, %7
  %11 = icmp samesign ugt i32 %8, 2
  br i1 %11, label %5, label %12, !llvm.loop !3

12:                                               ; preds = %5, %3
  %13 = phi i64 [ %4, %3 ], [ %10, %5 ]
  ret i64 %13
}
