package main

import "fmt"

func changeMe(slice []int) {
        slice[0] = 42
}

func main() {
        s := []int{1, 2, 3}
        fmt.Println(s)
        changeMe(s)
        fmt.Println(s)
}

2. 

package main

import "fmt"

type Employee struct {
        id    int
        name  string
        wages []int
}

func changeMe(employee Employee) {
        employee.id = -1
        employee.name = "Someone else"
        employee.wages[0] = -1
}

func main() {
        franta := Employee{
                id:    1,
                name:  "Franta",
                wages: []int{30000, 32000, 30500, 29900},
        }

        fmt.Println(franta)
        changeMe(franta)
        fmt.Println(franta)
}

package main

import "fmt"

const MONTHS = 12

type Employee struct {
        id    int
        name  string
        wages [MONTHS]int
}

func changeMe(employee Employee) {
        employee.id = -1
        employee.name = "Someone else"
        employee.wages[0] = -1
}

func main() {
        franta := Employee{
                id:    1,
                name:  "Franta",
                wages: [...]int{30000, 32000, 30500, 29900, 10000, 35000, 30000, 32000, 30500, 29900, 10000, 35000},
        }

        fmt.Println(franta)
        changeMe(franta)
        fmt.Println(franta)
}
