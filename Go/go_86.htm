import "C"
 
//export concat
func concat(text1, text2 *C.char) *C.char {
	t1 := C.GoString(text1)
	t2 := C.GoString(text2)
 
	result := t1 + t2
	return C.CString(result)
}
 
func main() {}

package main
 
// #include <stdlib.h>
import "C"
import "unsafe"
 
//export concat
func concat(text1, text2 *C.char) *C.char {
	t1 := C.GoString(text1)
	t2 := C.GoString(text2)
 
	result := t1 + t2
	return C.CString(result)
}
 
//export freeString
func freeString(s *C.char) {
	C.free(unsafe.Pointer(s))
}

func main() {}

3. 

import ctypes

so8 = ctypes.CDLL("./so8.so")

t1 = "ěščř ЩжΛλ".encode("utf-8")
t2 = "<foobar>".encode("utf-8")

so8.concat.restype = ctypes.POINTER(ctypes.c_char)

ptr = so8.concat(t1, t2)
val = ctypes.cast(ptr, ctypes.c_char_p).value
print(val.decode("utf-8"))
so8.freeString(ptr)

import ctypes
import time

so8 = ctypes.CDLL("./so8.so")

t1 = ("ěščř ЩжΛλ"*10000).encode("utf-8")
t2 = ("<foobar>"*10000).encode("utf-8")

so8.concat.restype = ctypes.POINTER(ctypes.c_char)

for i in range(100000):
    ptr = so8.concat(t1, t2)
    val = ctypes.cast(ptr, ctypes.c_char_p).value
    print(len(val))
    time.sleep(0.001)
