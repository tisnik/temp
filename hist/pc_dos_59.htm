        it      ne

        bx      lr

        subs    r0, r0, r3
        it      ne
        movne   r0, #1
        bx      lr

        lsrs    r0, r0, #16
        bx      lr
add_overflow_signed_int:
        adds    r0, r0, r1
        ite     vs
        movvs   r0, #1
        movvc   r0, #0
        bx      lr

        cmn     r0, r1
        ite     cs
        movcs   r0, #1

6. Překlad obecné funkce __builtin_add_overflow pro 64bitové ARMy (AArch64)

add_overflow_signed_char:
        sxtb    w1, w1
        sxtb    w0, w0
        add     w2, w0, w1
        eon     w0, w0, w1
        eor     w1, w1, w2
        and     w0, w1, w0
        ubfx    w0, w0, 7, 1
        ret
add_overflow_unsigned_char:
        and     w0, w0, 255
        add     w1, w0, w1
        cmp     w0, w1, uxtb
        cset    w0, hi
        ret
add_overflow_signed_short:
        sxth    w1, w1
        sxth    w0, w0
        add     w2, w0, w1
        eon     w0, w0, w1
        eor     w1, w1, w2
        and     x0, x1, x0
        ubfx    x0, x0, 15, 1
        ret
add_overflow_unsigned_short:
        and     w0, w0, 65535
        add     w1, w0, w1
        cmp     w0, w1, uxth
        cset    w0, hi
        ret
add_overflow_signed_int:
        cmn     w0, w1
        cset    w0, vs
        ret
add_overflow_unsigned_int:
        adds    w0, w0, w1
        cset    w0, cs
        ret
add_overflow_signed_long:
        cmn     x0, x1
        cset    w0, vs
        ret
add_overflow_unsigned_long:
        cmn     x0, x1
        cset    w0, cs
        ret
