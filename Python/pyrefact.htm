L = []
C = 100

for i in range(1001):
    L.append(i + C)

S = sum(L)
print(S)

c = 100
l = []

for i in range(1001):
    l.append(i + c)

s = sum(l)
print(s)

11. Částečná optimalizace

l = []

for i in range(1001):
    m = []
    for j in range(1001):
        m.append(j)
    l.extend(m)

s = sum(l)
print(s)

L = []

for _ in range(1001):
    m = list(range(1001))
    L.extend(m)

S = sum(L)
print(S)
